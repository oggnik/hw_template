\newcommand{\twopartdef}[4]
{
	\left\{
	\begin{array}{ll}
		#1 & \mbox{if } #2 \\
		#3 & \mbox{if } #4
	\end{array}
	\right.
}
\newcommand{\threepartdef}[6]
{
	\left\{
	\begin{array}{lll}
		#1 & \mbox{if } #2 \\
		#3 & \mbox{if } #4 \\
		#5 & \mbox{if } #6
	\end{array}
	\right.
}
\newcommand{\fourpartdef}[8]
{
	\left\{
	\begin{array}{lll}
		#1 & \mbox{if } #2 \\
		#3 & \mbox{if } #4 \\
		#5 & \mbox{if } #6 \\
		#7 & \mbox{if } #8
	\end{array}
	\right.
}
